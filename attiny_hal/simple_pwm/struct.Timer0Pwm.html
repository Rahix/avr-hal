Use TC0 for PWM (pins PB0, PB1)

§Example

let mut timer0 = Timer0Pwm::new(dp.TC0, Prescaler::Prescale64);

let mut d0 = pins.d0.into_output().into_pwm(&mut timer0);
let mut d1 = pins.d1.into_output().into_pwm(&mut timer0);

d0.set_duty(128);
d0.enable();
